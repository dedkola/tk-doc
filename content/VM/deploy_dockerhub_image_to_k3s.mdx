### Create a Deployment YAML without PVC

```yaml copy
apiVersion: apps/v1
kind: Deployment
metadata:
  name: postgres-deployment
  namespace: test-app
  labels:
    app: postgres
spec:
  replicas: 1
  selector:
    matchLabels:
      app: postgres
  template:
    metadata:
      labels:
        app: postgres
    spec:
      restartPolicy: Always
      containers:
        - name: postgres
          image: docker.io/dedkola/postgres:17
          imagePullPolicy: Always
          ports:
            - containerPort: 5432
          resources:
            requests:
              memory: "1Gi"
              cpu: "1000m"
            limits:
              memory: "2Gi"
              cpu: "2000m"

          volumeMounts:
            - name: postgres-storage
              mountPath: /var/lib/postgresql/data
      volumes:
        - name: postgres-storage
          emptyDir: {} # For production, use PersistentVolumeClaim
---
apiVersion: v1
kind: Service
metadata:
  name: postgres-service
  namespace: test-app
spec:
  type: LoadBalancer
  selector:
    app: postgres
  ports:
    - protocol: TCP
      port: 5432
      targetPort: 5432
```

### Check resources

```bash copy
kubectl top pod -n test-app
```

### Check logs

```bash copy
kubectl logs postgres-deployment-6577845884-8964z  -n test-app
```

### With PVC

```yaml copy
apiVersion: apps/v1
kind: Deployment
metadata:
  name: postgres-deployment
  namespace: test-app
  labels:
    app: postgres
spec:
  replicas: 1
  selector:
    matchLabels:
      app: postgres
  template:
    metadata:
      labels:
        app: postgres
    spec:
      restartPolicy: Always
      containers:
        - name: postgres
          image: docker.io/dedkola/postgres:17
          imagePullPolicy: Always
          ports:
            - containerPort: 5432
          resources:
            requests:
              memory: "1Gi"
              cpu: "1000m"
            limits:
              memory: "2Gi"
              cpu: "2000m"

          volumeMounts:
            - name: postgres-storage
              mountPath: /var/lib/postgresql/data
      volumes:
        - name: postgres-storage
          persistentVolumeClaim:
            claimName: postgres-pvc
---
apiVersion: v1
kind: Service
metadata:
  name: postgres-service
  namespace: test-app
spec:
  type: LoadBalancer
  selector:
    app: postgres
  ports:
    - protocol: TCP
      port: 5432
      targetPort: 5432
---
apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: postgres-pvc
  namespace: test-app
spec:
  accessModes:
    - ReadWriteOnce
  storageClassName: local-path
  resources:
    requests:
      storage: 10Gi
```
